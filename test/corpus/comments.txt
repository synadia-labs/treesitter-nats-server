===
Hash comment
===
# This is a comment
---
(source_file
  (comment))

===
Double slash comment
===
// This is a comment
---
(source_file
  (comment))

===
Multiple comments
===
# First comment
// Second comment
# Third comment
---
(source_file
  (comment)
  (comment)
  (comment))

===
Inline comment after pair
===
port: 4222 # default port
---
(source_file
  (pair
    key: (identifier)
    value: (number))
  (comment))

===
Double slash comments inside array
===
routes = [
  # hash comment
  // preceding comment
  nats-route://127.0.0.1:7246   // eol comment
]
---
(source_file
  (pair
    key: (identifier)
    value: (array
      (comment)
      (comment)
      (string
        (bare_string))
      (comment))))
